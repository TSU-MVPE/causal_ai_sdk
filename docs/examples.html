Examples

The SDK is used as an async context manager. All service calls go through client.kg, client.cd, or client.da.

Knowledge Graph

Create a session, upload a knowledge graph from file, list and inspect it:

async with CausalAIClient(api_key="...", base_url="...") as client:
    session = await client.kg.init_session()
    kg = await client.kg.upload_kg_from_file(
        session_uuid=session["uuid"],
        file_path="graph.json",
        title="My Graph",
    )
    kg_list = await client.kg.list_kg(session["uuid"])
    kg_detail = await client.kg.get_kg(session["uuid"], kg["id"])
    # When done: await client.kg.delete_session(session["uuid"])

Causal Discovery (MultiCa)

Upload multiple datasets with upload_data_for_multica (or use get_upload_url for a presigned URL and upload yourself), then run MultiCa with optional roots/sinks. Poll until complete and fetch the result:

async with CausalAIClient(api_key="...", base_url="...") as client:
    session = await client.kg.init_session()
    uploaded = await client.cd.upload_data_for_multica(
        session["uuid"], ["dataset1.csv", "dataset2.csv"]
    )
    task = await client.cd.run_multica(
        uploaded, threshold=0.01, roots=["X1"], sinks=["Y"]
    )
    status = await client.cd.wait_for_task(uploaded, timeout=300, interval=5)
    result = await client.cd.get_task_result(session["uuid"], task["task_id"])

Causal Discovery (LiNGAM)

Upload a single dataset, run LiNGAM causal discovery, wait for completion, and fetch the result:

async with CausalAIClient(api_key="...", base_url="...") as client:
    session = await client.kg.init_session()
    uploaded = await client.cd.upload_data_for_lingam(session["uuid"], "data.csv")
    task = await client.cd.run_lingam(uploaded, threshold=0.01)
    await client.cd.wait_for_task(uploaded, timeout=600, interval=5)
    task_result = await client.cd.get_task_result(session["uuid"], task["task_id"])
    # task_result contains result_url; fetch JSON from it for feature_names, etc.

Causal Discovery (TraCKR)

Upload a source knowledge graph, upload target data, run TraCKR matching, then run TraCKR CD with knowledge transfer. Wait for completion and fetch the result:

async with CausalAIClient(api_key="...", base_url="...") as client:
    session = await client.kg.init_session()
    kg = await client.kg.upload_kg_from_file(
        session_uuid=session["uuid"], file_path="source_graph.json"
    )
    uploaded = await client.cd.upload_data_for_trackr(session["uuid"], "target.csv")
    await client.cd.start_trackr_matching(uploaded, source_kg_id=kg["id"])
    matching = await client.cd.wait_for_matching(
        session["uuid"], mode="trackr", timeout=300, interval=3
    )
    # Optionally adjust: await client.cd.set_trackr_matching(session["uuid"], {...})
    task = await client.cd.run_trackr(
        uploaded,
        transferred_knowledge={"session_uuid": session["uuid"], "kg_id": kg["id"]},
        threshold=0.01,
    )
    await client.cd.wait_for_task(uploaded, timeout=300, interval=5)
    result = await client.cd.get_task_result(session["uuid"], task["task_id"])

Decision Analysis

Run explain or enumerate using a CD result reference. Poll until complete and fetch the result:

async with CausalAIClient(api_key="...", base_url="...") as client:
    session = await client.kg.init_session()
    # cd_result_reference = session_uuid + task_id from a completed CD task
    explain_task = await client.da.run_explain(
        session_uuid=session["uuid"],
        cd_result_reference={"session_uuid": "...", "task_id": "..."},
        current_observation={"feature_a": 0.5, "feature_b": 1.0},
        targets=[{"col": "target_col", "sense": ">", "threshold": 0.8}],
    )
    await client.da.wait_for_task(
        session["uuid"], explain_task["task_id"], timeout=300, interval=5
    )
    da_result = await client.da.get_task_result(
        session["uuid"], explain_task["task_id"]
    )
    # Or run_enumerate with same args for enumeration:
    # enum_task = await client.da.run_enumerate(...)

LiNGAM + Decision Analysis

Run causal discovery with LiNGAM, wait for completion, fetch the CD result, build DA input from it, then run Decision Analysis (explain). Full runnable script from the repository:

  1"""Example: LiNGAM causal discovery followed by Decision Analysis (explain).
  2
  3This example runs the full flow:
  41. Create session
  52. Upload data and run LiNGAM CD
  63. Wait for CD result and download it
  74. Build current_observation and targets from CD result (first row, first non-one-hot target)
  85. Submit DA explain task using CD result reference
  96. Wait for DA task and fetch result
 107. Cleanup
 11"""
 12
 13import asyncio
 14import math
 15import sys
 16from pathlib import Path
 17from typing import Any, Dict, List, Optional, Tuple
 18
 19import httpx
 20from causal_ai_sdk import CausalAIClient
 21from helpers import _delete_api_key, _get_api_key, _resolve_api_base_url

 24def _build_one_hot_set(cd_result: Dict[str, Any]) -> set:
 25    """Build set of one-hot feature names from CD result.
 26
 27    Args:
 28        cd_result (Dict[str, Any]): CD result dict with feature_names and feature_metadata.
 29
 30    Returns:
 31        Set of feature names that are one-hot encoded.
 32    """
 33    names = cd_result.get("feature_names") or []
 34    categories = (cd_result.get("feature_metadata") or {}).get("feature_categories") or []
 35    one_hot = {n for n in names if "=" in n}
 36    for group in categories:
 37        if isinstance(group, list):
 38            for idx in group:
 39                if isinstance(idx, int) and 0 <= idx < len(names):
 40                    one_hot.add(names[idx])
 41    return one_hot
 42
 43
 44def _is_nan(v: Any) -> bool:
 45    """Return True if value is NaN.
 46
 47    Args:
 48        v (Any): Value to check.
 49
 50    Returns:
 51        True if v is a float and NaN, else False.
 52    """
 53    return isinstance(v, float) and math.isnan(v)
 54
 55
 56def _build_current_observation_and_targets(
 57    cd_result: Dict[str, Any],
 58) -> Tuple[Dict[str, float], List[str]]:
 59    """Build current_observation (first row) and targets (first non-one-hot) from CD result.
 60
 61    One-hot: 0/1 or 0 if NaN; numeric: 0.0 if NaN else value.
 62    Targets: first non-one-hot feature name.
 63
 64    Args:
 65        cd_result (Dict[str, Any]): CD result with feature_names, training_data,
 66            feature_metadata.
 67
 68    Returns:
 69        Tuple of (current_observation dict, list of target names).
 70
 71    Raises:
 72        ValueError: If feature_names/training_data missing or no non-one-hot target.
 73    """
 74    names = cd_result.get("feature_names") or []
 75    rows = (cd_result.get("training_data") or {}).get("data") or []
 76    if not names or not rows:
 77        raise ValueError("CD result missing feature_names or training_data.data")
 78    first_row = rows[0]
 79    if len(first_row) != len(names):
 80        raise ValueError("First row length does not match feature_names")
 81
 82    one_hot = _build_one_hot_set(cd_result)
 83    obs: Dict[str, float] = {}
 84    for name, value in zip(names, first_row):
 85        if name in one_hot:
 86            obs[name] = int(value) if value in (0, 1, 0.0, 1.0, False, True) else 0
 87        else:
 88            obs[name] = 0.0 if _is_nan(value) else value
 89
 90    candidates = [n for n in names if n not in one_hot]
 91    targets = [candidates[0]] if candidates else []
 92    if not targets:
 93        raise ValueError("No non-one-hot feature for target")
 94    return obs, targets

 97async def run_lingam_da_workflow(
 98    client: CausalAIClient,
 99    test_data_dir: Path,
100) -> bool:
101    """Run LiNGAM then DA explain and return True if both succeed.
102
103    Args:
104        client (CausalAIClient): Causal AI client instance.
105        test_data_dir (Path): Path to directory containing trackr_target_data.json.
106
107    Returns:
108        True if both LiNGAM and DA workflows succeed, else False.
109    """
110    data_file = test_data_dir / "trackr_target_data.json"
111    if not data_file.exists():
112        print("  Test file not found (trackr_target_data.json), skipping...")
113        return False
114
115    session_uuid: Optional[str] = None
116    cd_task_id: Optional[str] = None
117    da_task_id: Optional[str] = None
118
119    try:
120        # --- Session ---
121        print("  Step 1: Create Session")
122        session = await client.kg.init_session()
123        session_uuid = session["uuid"]
124        print(f"  [OK] Session: {session_uuid}")
125
126        # --- LiNGAM CD ---
127        print("  Step 2: Upload Data and Run LiNGAM")
128        uploaded_data = await client.cd.upload_data_for_lingam(session_uuid, data_file)
129        cd_task_id = uploaded_data.task_id
130        print(f"  [OK] Uploaded, task_id: {cd_task_id}")
131
132        print("  Step 3: Submit LiNGAM Task")
133        await client.cd.run_lingam(uploaded_data, threshold=0.01)
134        print("  [OK] LiNGAM task submitted")
135
136        print("  Step 4: Wait for LiNGAM to Complete")
137        await client.cd.wait_for_task(uploaded_data, timeout=600, interval=5)
138        print("  [OK] LiNGAM completed")
139
140        print("  Step 5: Fetch CD Result")
141        task_result = await client.cd.get_task_result(session_uuid, cd_task_id)
142        if not task_result.get("result_url"):
143            print("  [ERROR] No CD result URL")
144            return False
145        async with httpx.AsyncClient() as http_client:
146            resp = await http_client.get(task_result.get("result_url"))
147            resp.raise_for_status()
148            cd_result = resp.json()
149        if cd_result.get("status") != "succeeded" or cd_result.get("mode") != "lingam":
150            print("  [ERROR] CD result not succeeded or not lingam mode")
151            return False
152        print("  [OK] CD result retrieved")
153
154        # --- Build DA input from CD result ---
155        print("  Step 6: Build DA Input from CD Result")
156        current_observation, targets = _build_current_observation_and_targets(cd_result)
157        cd_result_reference = {"session_uuid": session_uuid, "task_id": cd_task_id}
158        da_targets = [{"col": targets[0], "sense": ">", "threshold": 0.8}]
159        print(f"  [OK] current_observation keys: {len(current_observation)}, targets: {targets}")
160
161        # --- DA explain ---
162        print("  Step 7: Submit DA Explain Task")
163        da_task = await client.da.run_explain(
164            session_uuid=session_uuid,
165            cd_result_reference=cd_result_reference,
166            current_observation=current_observation,
167            targets=da_targets,
168        )
169        da_task_id = da_task["task_id"]
170        print(f"  [OK] DA task submitted: {da_task_id}")
171
172        print("  Step 8: Wait for DA Task to Complete")
173        await client.da.wait_for_task(session_uuid, da_task_id, timeout=300, interval=5)
174        print("  [OK] DA task succeeded")
175
176        print("  Step 9: Fetch DA Result")
177        da_result_resp = await client.da.get_task_result(session_uuid, da_task_id)
178        if not da_result_resp.get("result_url"):
179            print("  [ERROR] No DA result URL")
180            return False
181        async with httpx.AsyncClient() as http_client:
182            resp = await http_client.get(da_result_resp.get("result_url"))
183            resp.raise_for_status()
184            da_result = resp.json()
185        if da_result.get("status") != "succeeded":
186            print("  [ERROR] DA result status not succeeded")
187            return False
188        if "result" not in da_result:
189            print("  [ERROR] DA result missing 'result' key")
190            return False
191        print("  [OK] DA result retrieved")
192        print(f"    is_solved: {(da_result.get('result') or {}).get('is_solved')}")
193
194        return True
195
196    except Exception as e:
197        print(f"  [ERROR] Workflow failed: {e}")
198        return False
199
200    finally:
201        # Cleanup
202        print("  Step 10: Cleanup")
203        if session_uuid:
204            if da_task_id:
205                try:
206                    await client.da.delete_task(session_uuid, da_task_id)
207                    print("  [OK] DA task deleted")
208                except Exception as e:
209                    print(f"  [ERROR] (DA delete: {e})")
210            if cd_task_id:
211                try:
212                    await client.cd.delete_task(session_uuid, cd_task_id)
213                    print("  [OK] CD task deleted")
214                except Exception as e:
215                    print(f"  [ERROR] (CD delete: {e})")
216            try:
217                await client.kg.delete_session(session_uuid)
218                print("  [OK] Session deleted")
219            except Exception as e:
220                print(f"  [ERROR] (Session delete: {e})")
221

223async def main() -> None:
224    """Run LiNGAM + DA example."""
225    base_url = _resolve_api_base_url()
226    if not base_url:
227        print("\n[ERROR] Could not determine API Gateway URL. Set CAUSAL_AI_BASE_URL.")
228        sys.exit(1)
229
230    api_key, temp_company_name = _get_api_key()
231    if not api_key:
232        print("\n[ERROR] No API key. Set CAUSAL_AI_API_KEY or use cai-keymgr.")
233        sys.exit(1)
234
235    print("=" * 60)
236    print("LiNGAM + DA Example")
237    print("=" * 60)
238    print(f"Base URL: {base_url}")
239    print("=" * 60)
240
241    test_data_dir = Path(__file__).parent.parent.parent / "scripts" / "test_data" / "cd"
242
243    try:
244        async with CausalAIClient(api_key=api_key, base_url=base_url) as client:
245            success = await run_lingam_da_workflow(client, test_data_dir)
246        print("\n" + "=" * 60)
247        print("Summary: " + ("[OK] PASSED" if success else "[ERROR] FAILED/SKIPPED"))
248        print("=" * 60)
249    finally:
250        if temp_company_name:
251            _delete_api_key(temp_company_name)
252
253
254if __name__ == "__main__":
255    asyncio.run(main())

Column matching (MultiCa)

Upload data, start MultiCa column matching, wait for it to complete, then run MultiCa using the same session (and optionally set or inspect the matching).

from causal_ai_sdk.models import UploadedData

async with CausalAIClient(api_key="...", base_url="...") as client:
    session = await client.kg.init_session()
    upload_info = await client.cd.get_upload_url(session["uuid"])
    # ... upload your data to S3 using upload_info ...
    uploaded = UploadedData(
        task_id=upload_info.task_id,
        session_uuid=session["uuid"],
        s3_key=upload_info.s3_key,
    )
    await client.cd.start_multica_matching(uploaded)
    matching = await client.cd.wait_for_matching(session["uuid"], mode="multica")
    # Optionally: client.cd.set_multica_matching(session["uuid"], {"col_a": "src_a"})
    task = await client.cd.run_multica(uploaded, threshold=0.01)
